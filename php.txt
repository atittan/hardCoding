if문으로 처리할시

    if ($i == 1 ||
        $i == 2 ||
        $i == 3) {
     echo '$i is somewhere between 1 and 3.';
    }
    
switch 문으로 처리할 시

    case 1:
    case 2:
    case 3:
        echo '$i is somewhere between 1 and 3.';
        break;
    
    
@ endswitch
